```python
# src/utils/rl_env.py
"""Reinforcement-learning helpers for notebook 07.

Pure functions over numpy + a PortfolioEnv class. See
docs/superpowers/specs/2026-05-17-rl-agent-design.md for design.
"""
from __future__ import annotations

import numpy as np


def project_to_simplex(action: np.ndarray, max_weight: float = 0.10) -> np.ndarray:
    """Map a raw real-valued action vector to long-only weights.

    softmax -> clip per-element at `max_weight` -> renormalize to sum 1.
    """
    a = np.asarray(action, dtype=np.float64)
    a = a - a.max()  # numerical stability
    w = np.exp(a)
    w = w / w.sum()
    w = np.minimum(w, max_weight)
    s = w.sum()
    if s == 0:
        return np.full_like(w, 1.0 / len(w))
    return (w / s).astype(np.float32)
```

```python
TOP_FEATURES = ['payoutratio', 'ncfdiv', 'bidlo', 'sgna', 'retearn']
MACRO_COLS = ['macro_vixcls', 'macro_dgs10', 'macro_t10y2y']


def build_scoreboard_from_scored_panel(
    panel_df: pd.DataFrame,
    top_k: int = 30,
    date_col: str = 'date',
    score_col: str = 'score',
    target_col: str = 'fwd_ret_5d',
) -> pd.DataFrame:
    """Given a Friday-only panel with a pre-computed `score` column,
    keep top-K by score per date. Returns columns:
    [permno, date, score, fwd_ret_5d, *MACRO_COLS, *TOP_FEATURES]."""
    keep = ['permno', date_col, score_col, target_col, *MACRO_COLS, *TOP_FEATURES]
    df = panel_df[keep].copy()
    df = (df.sort_values([date_col, score_col], ascending=[True, False])
            .groupby(date_col, sort=False, group_keys=False)
            .head(top_k)
            .reset_index(drop=True))
    return df
```

```python
# src/utils/rl_env.py (append)
import gymnasium as gym
from gymnasium import spaces


OBS_DIM = 30 + 30 + 5 * 30 + 3 + 1  # 214


class PortfolioEnv(gym.Env):
    """Walk-1 portfolio allocation env.

    Each step: pick weights over top-K (already filtered by ranker score for
    this Friday), compute realized 5d return, advance to next Friday.

    Observation: weights (K) + scores (K, z-scored) + top-5 features (K*5, z-scored
    within date) + macro (3) + recent portfolio return (1).
    Action: Box(K), projected via softmax -> clip(max_weight) -> renorm.
    Reward: portfolio_return - (cost_bps / 1e4) * |new_weights - prev_weights|.sum().
    """

    metadata = {'render_modes': []}

    def __init__(
        self,
        scoreboard: pd.DataFrame,
        top_k: int = 30,
        episode_length: int = 52,
        cost_bps: float = 5.0,
        max_weight: float = 0.10,
    ):
        super().__init__()
        self.scoreboard = scoreboard.sort_values('date').reset_index(drop=True)
        self.top_k = top_k
        self.episode_length = episode_length
        self.cost_bps = float(cost_bps)
        self.max_weight = float(max_weight)

        # Group rows by date for O(1) per-Friday lookup.
        self._dates = np.array(sorted(self.scoreboard['date'].unique()))
        self._by_date: dict = {d: g.reset_index(drop=True)
                               for d, g in self.scoreboard.groupby('date')}

        self.action_space = spaces.Box(
            low=-np.inf, high=np.inf, shape=(top_k,), dtype=np.float32,
        )
        self.observation_space = spaces.Box(
            low=-np.inf, high=np.inf, shape=(OBS_DIM,), dtype=np.float32,
        )

    def reset(self, seed=None, options=None):
        super().reset(seed=seed)
        # Random start s.t. there are at least episode_length Fridays ahead.
        max_start = max(0, len(self._dates) - self.episode_length - 1)
        self._idx = int(self.np_random.integers(0, max_start + 1)) if max_start > 0 else 0
        self._steps = 0
        self._weights = np.full(self.top_k, 1.0 / self.top_k, dtype=np.float32)
        self._last_return = 0.0
        return self._build_obs(), {}

    def step(self, action: np.ndarray):
        cur = self._by_date[self._dates[self._idx]]
        new_weights = project_to_simplex(action, max_weight=self.max_weight)

        # Realized return over the next 5 trading days using current top-K's fwd_ret_5d.
        rets = cur['fwd_ret_5d'].to_numpy(dtype=np.float32)
        rets = np.nan_to_num(rets, nan=0.0)
        portfolio_return = float(np.dot(new_weights, rets[:self.top_k]))
        trade_amount = float(np.abs(new_weights - self._weights).sum())
        reward = portfolio_return - (self.cost_bps / 10_000.0) * trade_amount

        self._weights = new_weights
        self._last_return = portfolio_return
        self._idx += 1
        self._steps += 1
        terminated = self._steps >= self.episode_length
        # If we run out of dates mid-episode, terminate.
        if self._idx >= len(self._dates):
            terminated = True
        return self._build_obs(), reward, terminated, False, {
            'portfolio_return': portfolio_return,
            'trade_amount': trade_amount,
        }

    def _build_obs(self) -> np.ndarray:
        cur = self._by_date[self._dates[self._idx]]
        # 1. weights
        w = self._weights
        # 2. ranker scores, z-scored within date
        scores = cur['score'].to_numpy(dtype=np.float32)[:self.top_k]
        scores = (scores - scores.mean()) / (scores.std() + 1e-8)
        # 3. top-5 features per stock, z-scored within date
        feats = []
        for col in TOP_FEATURES:
            v = cur[col].to_numpy(dtype=np.float32)[:self.top_k]
            v = (v - np.nanmean(v)) / (np.nanstd(v) + 1e-8)
            feats.append(v)
        feats = np.concatenate(feats)  # 5 * top_k
        # 4. macro (take row 0; all rows on this date share the same macro)
        macro = cur[MACRO_COLS].iloc[0].to_numpy(dtype=np.float32)
        # 5. recent portfolio return
        recent = np.array([self._last_return], dtype=np.float32)

        obs = np.concatenate([w, scores, feats, macro, recent]).astype(np.float32)
        return np.nan_to_num(obs, nan=0.0, posinf=0.0, neginf=0.0)
```

```python
def _make_env_fn(scoreboard_subset: pd.DataFrame, cost_bps: float, seed: int):
    def _thunk():
        env = PortfolioEnv(
            scoreboard=scoreboard_subset,
            top_k=TOP_K,
            episode_length=EPISODE_LENGTH,
            cost_bps=cost_bps,
            max_weight=MAX_WEIGHT,
        )
        env.reset(seed=seed)
        return env
    return _thunk


def train_one(cost_bps: float, out_dir: Path) -> dict:
    out_dir.mkdir(parents=True, exist_ok=True)
    (out_dir / 'ckpts').mkdir(exist_ok=True)
    (out_dir / 'tb').mkdir(exist_ok=True)

    # Train vec env (4 parallel) + val env (1, deterministic).
    train_vec = DummyVecEnv([
        _make_env_fn(scoreboard_train, cost_bps, RANDOM_STATE + i)
        for i in range(N_ENVS)
    ])
    train_vec = VecNormalize(train_vec, norm_obs=True, norm_reward=False, clip_obs=10.0)

    val_vec = DummyVecEnv([_make_env_fn(scoreboard_val, cost_bps, RANDOM_STATE + 1000)])
    val_vec = VecNormalize(val_vec, norm_obs=True, norm_reward=False, clip_obs=10.0,
                           training=False)
    # Share normalization stats with train env at eval time (set via callback below).

    eval_cb = EvalCallback(
        val_vec,
        best_model_save_path=str(out_dir),
        log_path=str(out_dir),
        eval_freq=max(EVAL_FREQ // N_ENVS, 1),
        n_eval_episodes=1,
        deterministic=True,
    )
    ckpt_cb = CheckpointCallback(
        save_freq=max(CHECKPOINT_FREQ // N_ENVS, 1),
        save_path=str(out_dir / 'ckpts'),
        name_prefix='ppo',
    )

    model = PPO(
        'MlpPolicy',
        train_vec,
        policy_kwargs=dict(net_arch=[256, 128]),
        learning_rate=3e-4,
        n_steps=2048,
        batch_size=64,
        n_epochs=10,
        gamma=0.99,
        gae_lambda=0.95,
        clip_range=0.2,
        ent_coef=0.0,
        vf_coef=0.5,
        max_grad_norm=0.5,
        device='cpu',
        tensorboard_log=str(out_dir / 'tb'),
        seed=RANDOM_STATE,
        verbose=0,
    )

    t0 = time.time()
    model.learn(
        total_timesteps=TOTAL_TIMESTEPS,
        callback=[eval_cb, ckpt_cb, ProgressBarCallback()],
        progress_bar=False,  # ProgressBarCallback handles it
    )
    elapsed = time.time() - t0

    # Save final policy + normalization stats.
    model.save(out_dir / 'final_policy')
    train_vec.save(str(out_dir / 'vec_normalize.pkl'))

    # Pull EvalCallback's best.
    best_mean_reward = float(eval_cb.best_mean_reward)
    metrics = {
        'cost_bps': cost_bps,
        'total_timesteps': TOTAL_TIMESTEPS,
        'n_envs': N_ENVS,
        'wall_time_sec': elapsed,
        'wall_time_hr': elapsed / 3600.0,
        'best_val_mean_reward': best_mean_reward,
    }
    (out_dir / 'training_metrics.json').write_text(json.dumps(metrics, indent=2))
    print(f'  cost={cost_bps}bps done; wall={elapsed/60:.1f} min; '
          f'best_val_mean_reward={best_mean_reward:.4f}')
    return metrics
```
